import React, { useState, useEffect } from "react";
import { Stepper, Step, StepLabel, Button } from "@mui/material";
import { ToastContainer, toast } from "react-toastify";
import "react-toastify/dist/ReactToastify.css";
import { useNavigate } from "react-router-dom";
import { IBooking } from "../interface";
import { useSelector } from "react-redux";
 
// Stepper labels as an array of strings
const steps = ["Pending", "Approved", "Completed"];
 
const BookingHistory: React.FC = () => {
  const navigate = useNavigate();
  // Fetch booking data from the Redux store
  const data = useSelector((state: any) => state.bookingReducer.booking);
  console.log(data);
 
  // Initialize bookings state with data from the saga store
  const [bookings, setBookings] = useState<IBooking[]>(data);
 
  useEffect(() => {
    // Update bookings state when data changes
    setBookings(data);
  }, [data]);
 
  // Remove the booking from the state
  const handleCancelRequest = (id: string) => {
    setBookings(bookings.filter((booking) => booking._id !== id));
    toast.success("Booking cancelled");
  };
 
  // Navigate back to the cab booking page
  const handleBack = () => {
    navigate("/cab");
  };
 
  return (
    <div className="flex flex-col p-4">
      {bookings.map((booking, index) => (
        <div
          key={index}
          className="bg-white p-4 rounded-lg shadow-md m-2 relative"
        >
          {/* Display booking details */}
          <p className="text-lg font-bold">{booking.date}</p>
          <p className="text-md text-gray-600">
            {booking.cabType} - Radha Nanjundaswamy(51422999)
          </p>
          <p className="text-sm">{booking.pickupLocation}</p>
          <p className="text-sm mb-4">{booking.dropLocation}</p>
          {/* Display booking status using a stepper */}
          <Stepper activeStep={booking.status} alternativeLabel>
            {steps.map((label, stepIndex) => (
              <Step key={stepIndex}>
                <StepLabel>{label}</StepLabel>
              </Step>
            ))}
          </Stepper>
          {/* Button to cancel the booking */}
          <Button
            variant="contained"
            color="primary"
            className="absolute top-0 right-0 mt-2 mr-2"
            onClick={() => handleCancelRequest(booking._id)}
          >
            Cancel Request
          </Button>
        </div>
      ))}
      {/* Button to navigate back */}
      <div className="flex justify-center mt-4">
        <Button variant="contained" color="secondary" onClick={handleBack}>
          Back
        </Button>
      </div>
      <ToastContainer />
    </div>
  );
};
 
export default BookingHistory;
